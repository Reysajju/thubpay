  ThubPay — Multi-Gateway Payment Platform
                                                                                                                                                                                                                                                                                 What It Is
                                                                                                                                                                                                                                                                                 ThubPay is a Next.js-based multi-gateway payment portal designed for startups. It's a SaaS platform that lets businesses (brands) accept payments, manage invoices, and handle subscriptions through multiple payment gateways from a single dashboard.

  What's Built (Phase 1)

  1. Multi-tenant architecture — Workspaces, workspace members, brands, clients
  2. Stripe integration (complete) — Full adapter: customers, payment methods, subscriptions, refunds, webhooks
  3. Gateway abstraction layer — PaymentGateway interface + registry/factory pattern, ready for 9 gateways (Stripe, PayPal, Square, Adyen, Razorpay, Authorize.net, Mollie, 2Checkout, Klarna)
  4. RBAC — Owner, admin, viewer, member, billing roles with permission checks
  5. Audit logging — Every charge attempt is logged with user, action, and payload
  6. Payment router — Resolves which gateway to use per-tenant (defaults to Stripe)
  7. Dashboard — MRR, incoming/outgoing, profit/loss cards, brand/client/invoice/payment link management
  8. Invoice lifecycle — Create, send, view, payment link generation via Stripe Checkout
  9. Landing pages — About, blogs, contact, FAQs, how-it-works, privacy, terms, security, pricing
  10. DB schema — Brands, clients, addresses, gateway customers, payment methods, products, prices, invoices, transactions, webhook events — all with Row Level Security
  11. Webhook deduplication — Idempotency keys on webhook events table

  400 CREATE OR REPLACE FUNCTION mark_onboarding_step_completed()
  401 RETURNS trigger LANGUAGE plpgsql SECURITY DEFINER AS $$
  402 DECLARE
  403     ws_id   uuid;
  404     step_nm text;
  405 BEGIN
  406     CASE TG_TABLE_NAME
  407         WHEN 'clients'             THEN step_nm := 'add_client';
  408         WHEN 'invoices'            THEN step_nm := 'create_invoice';
  409         WHEN 'payment_links'       THEN step_nm := 'create_payment_link';
  410         WHEN 'gateway_credentials' THEN step_nm := 'connect_gateway';
  411         WHEN 'brands'              THEN step_nm := 'customize_brand';
  412         ELSE RETURN NEW;
  413     END CASE;
  414
  415     BEGIN
  416         ws_id := NEW.workspace_id;
  417     EXCEPTION WHEN undefined_column THEN
  418         RETURN NEW;
  419     END;
  420
  421     INSERT INTO onboarding_steps (workspace_id, step_name, is_completed, completed_at)
  422     VALUES (ws_id, step_nm, true, now())
  423     ON CONFLICT (workspace_id, step_name)
  424     DO UPDATE SET is_completed = true, completed_at = now();
  425
  426     IF NOT EXISTS (
  427         SELECT 1 FROM onboarding_steps WHERE workspace_id = ws_id AND is_completed = false
  428     ) AND EXISTS (
  429         SELECT 1 FROM onboarding_steps WHERE workspace_id = ws_id
  430     ) THEN
  431         UPDATE workspaces SET onboarding_completed = true WHERE id = ws_id AND onboarding_completed = false;
  432     END IF;
  433
  434     RETURN NEW;
  435 END;
  436 $$;
  437
  438 DO $$
  439 DECLARE
  440     trg record;
  441 BEGIN
  442     FOR trg IN SELECT * FROM (VALUES
  443         ('clients','trg_onboarding_client'),
  444         ('invoices','trg_onboarding_invoice'),
  445         ('payment_links','trg_onboarding_payment_link'),
  446         ('gateway_credentials','trg_onboarding_gateway'),
  447         ('brands','trg_onboarding_brand')
  448     ) AS t(tbl, trg_name)
  449     LOOP
  450         EXECUTE format('DROP TRIGGER IF EXISTS %I ON %I', trg.trg_name, trg.tbl);
  451         EXECUTE format(
  452             'CREATE TRIGGER %I AFTER INSERT ON %I FOR EACH ROW EXECUTE FUNCTION mark_onboarding_step_completed()',
  453             trg.trg_name, trg.tbl
  454         );
  455     END LOOP;
  456 END$$;
